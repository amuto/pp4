g_signal_connect(ウィジェット, シグナル, G_CALLBACK(関数), NULL);
第1引数のウィジェットで第2引数のシグナルが起きたら第3引数のコールバック関数を実行するように設定する
シグナルは特定の文字列
使用例（変数wが存在すると仮定）:
g_signal_connect(w, "destroy", G_CALLBACK(gtk_main_quit), NULL);
